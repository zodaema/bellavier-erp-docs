## ðŸ­ Backend Implementation

### Injection Flow

```php
class ComponentInjectionService
{
    public function injectMissingComponent(
        int $parentTokenId,
        string $componentCode,
        int $requestedBy
    ): array {
        // Step 1: Validate
        $this->validateInjection($parentTokenId, $componentCode);
        
        // Step 2: Check if already exists (idempotent)
        $existing = $this->findExistingToken($parentTokenId, $componentCode);
        if ($existing) {
            return [
                'token' => $existing,
                'already_exists' => true
            ];
        }
        
        // Step 3: Create component token
        $token = $this->createComponentToken($parentTokenId, $componentCode);
        
        // Step 4: Start routing (use standard component workflow)
        $firstNode = $this->getFirstNodeForComponent($componentCode);
        $this->routeTokenToNode($token['id'], $firstNode['id']);
        
        // Step 5: Log injection event
        $this->logInjectionEvent($parentTokenId, $componentCode, $token['id'], $requestedBy);
        
        // Step 6: Notify merge node to wait
        $this->notifyMergeNodeToWait($parentTokenId, $componentCode);
        
        return [
            'token' => $token,
            'next_node' => $firstNode,
            'already_exists' => false
        ];
    }
    
    private function validateInjection(int $parentTokenId, string $componentCode): void
    {
        // 1. Parent token must exist and not be scrapped
        $parent = $this->tokenRepo->find($parentTokenId);
        if (!$parent || $parent['status'] === 'scrapped') {
            throw new ComponentInjectionException("Parent token is invalid or scrapped");
        }
        
        // 2. Component code must be in catalog
        if (!$this->catalogService->isValidCode($componentCode)) {
            throw new ComponentInjectionException("Component code not in catalog");
        }
        
        // 3. Component must be valid for this product
        $productId = $parent['product_id'];
        if (!$this->isComponentValidForProduct($componentCode, $productId)) {
            throw new ComponentInjectionException("Component not valid for this product");
        }
    }
}
```

```php
class MergeNodeService
{
    public function checkMergeReadiness(int $mergeNodeId, int $parentTokenId): array
    {
        $expectedComponents = $this->getExpectedComponents($mergeNodeId);
        $existingTokens = $this->getComponentTokens($parentTokenId);
        
        $status = [
            'ready' => true,
            'waiting_for' => [],
            'completed' => []
        ];
        
        foreach ($expectedComponents as $componentCode) {
            $token = $existingTokens[$componentCode] ?? null;
            
            if (!$token) {
                $status['ready'] = false;
                $status['waiting_for'][] = [
                    'component' => $componentCode,
                    'reason' => 'missing_token'
                ];
            } elseif ($token['status'] !== 'completed') {
                $status['ready'] = false;
                $status['waiting_for'][] = [
                    'component' => $componentCode,
                    'reason' => 'in_progress',
                    'current_node' => $token['current_node_code']
                ];
            } else {
                $status['completed'][] = $componentCode;
            }
        }
        
        return $status;
    }
}
```

### Dashboard Query

```sql
SELECT 
    p.product_code,
    cil.component_code,
    COUNT(*) as injection_count,
    AVG(TIMESTAMPDIFF(HOUR, cil.created_at, t.completed_at)) as avg_completion_hours
FROM component_injection_log cil
JOIN flow_token pt ON pt.id = cil.parent_token_id
JOIN product p ON p.id = pt.product_id
JOIN flow_token t ON t.id = cil.created_token_id
WHERE cil.created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
GROUP BY p.product_code, cil.component_code
ORDER BY injection_count DESC;
```
